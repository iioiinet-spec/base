컴퓨터를 정점, 연결을 간선으로 보면 1번에서 DFS로 도달 가능한 모든 정점을 세면 된다. 1번 자신은 제외.

// DFS 패턴
그래프 탐색의 가장 기본 패턴. visited 배열로 중복 방문을 막고, 재귀로 연결된 정점을 모두 방문한다.
풀이 단계
1
그래프 구성: graph[a][b] = graph[b][a] = 1로 무방향 그래프를 인접 행렬로 저장.
2
DFS 시작: dfs(1) 호출. 전역 변수 cnt = 0으로 초기화.
3
재귀 탐색: visited[x]=1 표시 후, 1~n까지 graph[x][i]가 1이고 미방문이면 cnt++ 후 재귀.
4
왜 cnt++를 재귀 전에? dfs가 실제로 방문할 때마다 카운트. 1번 자신은 cnt에 포함 안 됨.
C · BOJ 2606
#include <stdio.h>

int graph[101][101];
int visited[101];
int n, m, cnt = 0;

void dfs(int x) {
    visited[x] = 1;
    for (int i = 1; i <= n; i++) {
        if (graph[x][i] && !visited[i]) {
            cnt++;           // 새 정점 방문
            dfs(i);
        }
    }
}

int main() {
    scanf("%d", &n);
    scanf("%d", &m);

    for (int i = 0; i < m; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        graph[a][b] = 1;
        graph[b][a] = 1;
    }

    dfs(1);
    printf("%d\n", cnt);
    return 0;
}
// DFS 그래프 탐색 시각화
방문: 0 | 스택 깊이: 0

2
DFS 재귀: visited[x]=1 후 출력. 1~n 순서로 연결된 미방문 정점을 재귀 호출.
3
작은 번호 우선: for(i=1; i<=n; i++) 루프가 자동으로 작은 번호부터 탐색하게 만든다.
DFS 순서 추적 (예제):
1 방문 → 1과 연결: 2,3,4 중 2가 첫번째 → 2 방문 → 2와 연결: 1(방문됨),4 중 4 → 4 방문 → 4와 연결: 1,2(방문됨),3 → 3 방문
결과: 1 2 4 3
C · BOJ 1260 (DFS부분)
#include <stdio.h>
#include <string.h>

int graph[1001][1001];
int visited[1001];
int n, m, v;

void dfs(int x) {
    visited[x] = 1;
    printf("%d ", x);    // 방문 즉시 출력

    for (int i = 1; i <= n; i++) {
        if (graph[x][i] && !visited[i]) {
            dfs(i);         // 작은 번호부터 자동
        }
    }
}

int main() {
    scanf("%d %d %d", &n, &m, &v);

    for (int i = 0; i < m; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        graph[a][b] = 1;
        graph[b][a] = 1;
    }

    dfs(v);
    printf("\n");
    return 0;
}

// CORE IDEA
DFS에 깊이(depth) 매개변수를 추가. 목표 정점에 도달했을 때 depth가 바로 촌수.
1
depth 파라미터: dfs(a, 0)으로 시작. 재귀마다 dfs(i, depth+1).
2
목표 도달: if(x == b)이면 answer = depth로 저장하고 return.
3
answer = -1: 전역 변수로 -1로 초기화. b에 도달 못하면 그대로 -1 출력.
4
7 → 3 경로: 7→2→1→3 순으로 탐색. depth가 0→1→2→3이 되어 답은 3.
C · BOJ 2644
#include <stdio.h>

int graph[101][101];
int visited[101];
int answer = -1;
int n, a, b, m;

void dfs(int x, int depth) {
    if (x == b) {
        answer = depth;     // 목표 도달!
        return;
    }
    visited[x] = 1;
    for (int i = 1; i <= n; i++) {
        if (graph[x][i] && !visited[i]) {
            dfs(i, depth + 1);  // 깊이 +1
        }
    }
}

int main() {
    scanf("%d", &n);
    scanf("%d %d", &a, &b);
    scanf("%d", &m);

    for (int i = 0; i < m; i++) {
        int x, y;
        scanf("%d %d", &x, &y);
        graph[x][y] = 1;
        graph[y][x] = 1;
    }

    dfs(a, 0);
    printf("%d\n", answer);
    return 0;
}
// 가족 트리 DFS — 7번 → 3번 촌수

그림

도화지에서 1이 상하좌우로 연결된 그림의 수와 가장 넓은 그림의 넓이를 구하라.

// CORE IDEA
DFS가 반환값을 가진다는 점이 핵심. 한 번의 DFS로 덩어리 전체의 크기(area)를 계산해서 반환.
1
반환값 있는 DFS: int dfs(x, y) — 각 재귀 호출이 area를 1씩 더해 반환.
2
4방향 이동: dx={1,-1,0,0}, dy={0,0,1,-1}로 상하좌우 탐색.
3
경계 체크: nx, ny가 0~n-1, 0~m-1 범위 내여야 함. 범위 초과 시 continue.
4
count vs maxArea: DFS 시작 횟수 → 그림 수. DFS 반환값 최대 → 가장 넓은 넓이.
C · BOJ 1926
#include <stdio.h>

int n, m;
int board[501][501];
int visited[501][501];
int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};

int dfs(int x, int y) {
    visited[x][y] = 1;
    int area = 1;   // 현재 칸 포함

    for (int d = 0; d < 4; d++) {
        int nx = x + dx[d];
        int ny = y + dy[d];
        if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
        if (!visited[nx][ny] && board[nx][ny] == 1) {
            area += dfs(nx, ny);   // 재귀 반환값 누적
        }
    }
    return area;
}

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            scanf("%d", &board[i][j]);

    int count = 0, maxArea = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (board[i][j] == 1 && !visited[i][j]) {
                int area = dfs(i, j);
                count++;
                if (area > maxArea) maxArea = area;
            }
        }
    }
    printf("%d\n%d\n", count, maxArea);
    return 0;
}
// 2D 격자 DFS — 그림 넓이 계산
그림: 0 | 최대: 0
격자에서 연결된 1 덩어리를 DFS로 탐색합니다.

전쟁 — 전투

같은 팀 병사가 연결된 무리 크기의 제곱이 위력. 아군(W)과 적군(B)의 총 위력을 구하라.

// CORE IDEA
그림(1926)과 동일 패턴인데 같은 문자(팀)끼리만 연결. DFS 파라미터에 team 문자를 추가.
1
team 파라미터: dfs(x, y, board[x][y])로 호출. 같은 문자인 칸만 탐색.
2
visited 체크: 모든 칸에서 미방문이면 DFS 실행. W든 B든 상관없이.
3
제곱 누적: dfs가 반환한 size를 size*size로 해당 팀(W/B)에 더함.
4
입력 주의: scanf("%d %d", &n, &m)에서 n=열수, m=행수. 루프는 m행 × n열.
C · BOJ 1303
#include <stdio.h>

int n, m;
char board[101][101];
int visited[101][101];
int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};

int dfs(int x, int y, char team) {
    visited[x][y] = 1;
    int cnt = 1;

    for (int d = 0; d < 4; d++) {
        int nx = x + dx[d], ny = y + dy[d];
        if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
        if (!visited[nx][ny] && board[nx][ny] == team) {
            cnt += dfs(nx, ny, team);
        }
    }
    return cnt;
}

int main() {
    scanf("%d %d", &n, &m);
    for (int i = 0; i < m; i++) scanf("%s", board[i]);

    int white = 0, blue = 0;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (!visited[i][j]) {
                int size = dfs(i, j, board[i][j]);
                if (board[i][j] == 'W') white += size * size;
                else blue += size * size;
            }
        }
    }
    printf("%d %d\n", white, blue);
    return 0;
}
// 전쟁 격자 DFS — 팀별 무리 계산
W: 0 | B: 0
팀별 무리를 DFS로 탐색하며 위력을 계산합니다.

현수막

1이 상하좌우 + 대각선 8방향으로 연결되면 한 글자. 글자 개수를 구하라.

// CORE IDEA
4방향 DFS와 완전히 동일한 구조인데 방향 수만 8개. dx/dy 배열에 대각선 방향 4개 추가.
1
8방향 배열: dx={1,-1,0,0,1,1,-1,-1}, dy={0,0,1,-1,1,-1,1,-1}
2

3

주의! 4방향인지 8방향인지 문제를 꼭 확인해야 한다. 이 문제에서 4방향으로 풀면 글자 수가 더 많이 나온다.
C · BOJ 14716
#include <stdio.h>

int m, n;
int board[251][251];
int visited[251][251];
// 8방향 — 대각선 포함!
int dx[8] = {1,-1,0,0,1,1,-1,-1};
int dy[8] = {0,0,1,-1,1,-1,1,-1};

void dfs(int x, int y) {
    visited[x][y] = 1;

    for (int d = 0; d < 8; d++) {   // 8방향
        int nx = x + dx[d];
        int ny = y + dy[d];
        if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
        if (!visited[nx][ny] && board[nx][ny] == 1) {
            dfs(nx, ny);
        }
    }
}

int main() {
    scanf("%d %d", &m, &n);
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &board[i][j]);

    int count = 0;
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            if (board[i][j] == 1 && !visited[i][j]) {
                dfs(i, j);
                count++;
            }

    printf("%d\n", count);
    return 0;
}

헌내기는 친구가 필요해

캠퍼스에서 도연이(I)가 벽(X)을 피해 만날 수 있는 사람(P)의 수를 구하라.

📋 문제
🔍 풀이
💻 코드
🎮 시각화
✏️ 퀴즈
입출력 예제
입력
3 5
OOOPO
OIOOX
OOOXP
출력
1
// CORE IDEA
시작점 I에서 DFS. X만 막히고 O, P, I 모두 이동 가능. 방문 중 P를 만나면 cnt++. 0이면 "TT" 출력.
1
시작점 찾기: 입력 받으면서 campus[i][j]=='I'이면 sx=i, sy=j 저장.
2
이동 조건: campus[nx][ny] != 'X'만 확인. O, P, I 모두 이동 가능.
3
P 카운트: DFS 내에서 if(campus[x][y]=='P') cnt++로 방문한 P를 셈.
4
TT 출력: DFS 후 cnt==0이면 "TT", 아니면 cnt 출력.
C · BOJ 21736
#include <stdio.h>

int n, m;
char campus[601][601];
int visited[601][601];
int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};
int cnt = 0;

void dfs(int x, int y) {
    visited[x][y] = 1;
    if (campus[x][y] == 'P') cnt++;  // 친구 발견!

    for (int d = 0; d < 4; d++) {
        int nx = x + dx[d];
        int ny = y + dy[d];
        if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
        if (!visited[nx][ny] && campus[nx][ny] != 'X') {
            dfs(nx, ny);
        }
    }
}

int main() {
    scanf("%d %d", &n, &m);
    int sx = 0, sy = 0;
    for (int i = 0; i < n; i++) {
        scanf("%s", campus[i]);
        for (int j = 0; j < m; j++)
            if (campus[i][j] == 'I') { sx = i; sy = j; }
    }

    dfs(sx, sy);

    if (cnt == 0) printf("TT\n");
    else printf("%d\n", cnt);
    return 0;
}
// 캠퍼스 DFS — 친구 찾기
친구: 0명
도연이(I)가 벽(X)을 피해 친구(P)를 찾습니다.
