// CORE IDEA배추밭에서 연결된 배추 덩어리 개수 = DFS 시작 횟수. 전형적인 연결 요소 세기 패턴. 테스트케이스마다 memset 필수!
// 주의: 좌표 역전입력이 (x, y) = (열, 행) 순서로 주어짐 → board[y][x] = 1 로 저장해야 함!
01

02
좌표 주의: scanf로 x(열), y(행) 입력 → board[y][x] = 1. 행/열 반전 실수 잦음!
03
연결 요소 세기: board[i][j]==1이고 미방문이면 DFS 실행 → count++. DFS 시작 횟수 = 덩어리 수.
04
시간복잡도: O(T × N × M). N, M ≤ 50이므로 충분히 빠름.
C · BOJ 1012
#include <stdio.h>
#include <string.h>

int board[51][51], visited[51][51];
int dx[4]={1,-1,0,0}, dy[4]={0,0,1,-1};
int n, m;

void dfs(int x, int y) {
    visited[x][y] = 1;
    for (int d = 0; d < 4; d++) {
        int nx = x+dx[d], ny = y+dy[d];
        if (nx<0||nx>=n||ny<0||ny>=m) continue;
        if (!visited[nx][ny] && board[nx][ny]==1)
            dfs(nx, ny);
    }
}

int main() {
    int T; scanf("%d", &T);
    while (T--) {
        int k;
        scanf("%d %d %d", &m, &n, &k);
        memset(board,0,sizeof(board));
        memset(visited,0,sizeof(visited));
        for (int i=0; i<k; i++) {
            int x,y; scanf("%d %d",&x,&y);
            board[y][x] = 1;  // 행/열 주의!
        }
        int count=0;
        for (int i=0; i<n; i++)
            for (int j=0; j<m; j++)
                if (board[i][j]==1 && !visited[i][j]) {
                    dfs(i,j); count++;
                }
        printf("%d\n", count);
    }
    return 0;
}

C · BOJ 2667
#include <stdio.h>
#include <stdlib.h>

char board[26][26];
int visited[26][26], result[626], n;
int dx[4]={1,-1,0,0}, dy[4]={0,0,1,-1};

int dfs(int x, int y) {
    visited[x][y] = 1;
    int cnt = 1;  // 현재 칸 포함
    for (int d=0; d<4; d++) {
        int nx=x+dx[d], ny=y+dy[d];
        if (nx<0||nx>=n||ny<0||ny>=n) continue;
        if (!visited[nx][ny] && board[nx][ny]=='1')
            cnt += dfs(nx, ny);  // 재귀 반환값 누적
    }
    return cnt;
}

int cmp(const void*a,const void*b){ return (*(int*)a-*(int*)b); }

int main() {
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%s",board[i]);
    int total=0;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(board[i][j]=='1'&&!visited[i][j])
                result[total++]=dfs(i,j);
    qsort(result,total,sizeof(int),cmp);
    printf("%d\n",total);
    for(int i=0;i<total;i++) printf("%d\n",result[i]);
    return 0;
}

01

03

04

C · BOJ 11724
#include <stdio.h>

int graph[1001][1001], visited[1001];
int n, m;

void dfs(int x) {
    visited[x] = 1;
    for (int i=1; i<=n; i++)
        if (graph[x][i] && !visited[i])
            dfs(i);
}

int main() {
    scanf("%d %d", &n, &m);
    for (int i=0; i<m; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        graph[a][b] = graph[b][a] = 1;
    }
    int count = 0;
    for (int i=1; i<=n; i++)
        if (!visited[i]) { dfs(i); count++; }
    printf("%d\n", count);
    return 0;
}

#include <stdio.h>
#include <string.h>

int board[51][51], visited[51][51];
int w, h;
// 8방향!
int dx[8]={1,-1,0,0,1,1,-1,-1};
int dy[8]={0,0,1,-1,1,-1,1,-1};

void dfs(int x, int y) {
    visited[x][y] = 1;
    for (int d=0; d<8; d++) {  // 8방향 탐색
        int nx=x+dx[d], ny=y+dy[d];
        if (nx<0||nx>=h||ny<0||ny>=w) continue;
        if (!visited[nx][ny] && board[nx][ny]==1)
            dfs(nx, ny);
    }
}

int main() {
    while(1) {
        scanf("%d %d", &w, &h);
        if (w==0 && h==0) break;
        memset(board,0,sizeof(board));
        memset(visited,0,sizeof(visited));
        for(int i=0;i<h;i++)
            for(int j=0;j<w;j++)
                scanf("%d",&board[i][j]);
        int count=0;
        for(int i=0;i<h;i++)
            for(int j=0;j<w;j++)
                if(board[i][j]==1&&!visited[i][j]) { dfs(i,j); count++; }
        printf("%d\n", count);
    }
    return 0;
}

// CORE IDEA높이 h를 1부터 max까지 바꾸며 매번 DFS. dfs 조건: arr[nx][ny] > h. 각 높이에서 DFS 시작 횟수 = 안전 영역 수 → 최댓값 출력.
01
브루트포스: h = 1 ~ maxH까지 모든 높이 시뮬레이션. N≤100이므로 O(N³)도 충분.
02

03
DFS 조건 변경: board[nx][ny] > h인 칸만 탐색. h보다 높은 곳만 안전.
04
answer=1: h=0일 때(아무것도 안 잠겼을 때) 전체가 1개의 영역. 이걸 초기값으로.
C · BOJ 2468
#include <stdio.h>
#include <string.h>

int n, board[101][101], visited[101][101];
int dx[4]={1,-1,0,0}, dy[4]={0,0,1,-1};

void dfs(int x, int y, int h) {
    visited[x][y]=1;
    for(int d=0;d<4;d++){
        int nx=x+dx[d], ny=y+dy[d];
        if(nx<0||nx>=n||ny<0||ny>=n) continue;
        if(!visited[nx][ny] && board[nx][ny]>h)
            dfs(nx,ny,h);  // h보다 높은 칸만!
    }
}

int main() {
    scanf("%d",&n);
    int maxH=0;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++){
            scanf("%d",&board[i][j]);
            if(board[i][j]>maxH) maxH=board[i][j];
        }
    int answer=1;
    for(int h=1;h<=maxH;h++){
        memset(visited,0,sizeof(visited));
        int count=0;
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                if(board[i][j]>h&&!visited[i][j]){
                    dfs(i,j,h); count++;
                }
        if(count>answer) answer=count;
    }
    printf("%d\n",answer);
    return 0;
}
// 수위별 안전 영역 시뮬레이션

#include <stdio.h>
#include <stdlib.h>

int board[101][101], visited[101][101];
int m, n, k;
int dx[4]={1,-1,0,0}, dy[4]={0,0,1,-1};
int area[10001];

int dfs(int x, int y) {
    visited[x][y]=1; int cnt=1;
    for(int d=0;d<4;d++){
        int nx=x+dx[d], ny=y+dy[d];
        if(nx<0||nx>=m||ny<0||ny>=n) continue;
        if(!visited[nx][ny]&&board[nx][ny]==0)
            cnt+=dfs(nx,ny);  // 0인 칸 탐색
    }
    return cnt;
}

int cmp(const void*a,const void*b){ return (*(int*)a-*(int*)b); }

int main() {
    scanf("%d %d %d",&m,&n,&k);
    for(int i=0;i<k;i++){
        int x1,y1,x2,y2;
        scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
        for(int y=y1;y<y2;y++)
            for(int x=x1;x<x2;x++)
                board[y][x]=1;
    }
    int count=0;
    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
            if(!board[i][j]&&!visited[i][j])
                area[count++]=dfs(i,j);
    qsort(area,count,sizeof(int),cmp);
    printf("%d\n",count);
    for(int i=0;i<count;i++) printf("%d ",area[i]);
    printf("\n");
    return 0;
}

// 인덱스 주의입력이 1-indexed! board[r][c] = 1. 경계 체크: nx<1 || nx>n || ny<1 || ny>m
01
1-indexed: board[r][c] = 1로 저장. 루프는 i=1~n, j=1~m. 경계 체크도 1 기준.
02

04
예제 분석: (1,1), (2,2), (2,3), (3,1), (3,2)가 연결 → 4칸 덩어리 + 1칸 덩어리 → 정답 4.
C · BOJ 1743
#include <stdio.h>

int board[101][101], visited[101][101];
int n, m, k;
int dx[4]={1,-1,0,0}, dy[4]={0,0,1,-1};

int dfs(int x, int y) {
    visited[x][y]=1; int cnt=1;
    for(int d=0;d<4;d++){
        int nx=x+dx[d], ny=y+dy[d];
        if(nx<1||nx>n||ny<1||ny>m) continue;  // 1-indexed!
        if(!visited[nx][ny]&&board[nx][ny]==1)
            cnt+=dfs(nx,ny);
    }
    return cnt;
}

int main() {
    scanf("%d %d %d",&n,&m,&k);
    for(int i=0;i<k;i++){
        int r,c; scanf("%d %d",&r,&c);
        board[r][c]=1;
    }
    int ans=0;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(board[i][j]==1&&!visited[i][j]){
                int s=dfs(i,j);
                if(s>ans) ans=s;
            }
    printf("%d\n",ans);
    return 0;
}
